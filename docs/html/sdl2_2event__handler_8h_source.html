39bool initializeEventManagers()
40{
41 // Initialize Key Event Manager
42 if (!initializeEventManager(&gKeyPressEventManager, MAX_KEY_LISTENER_LISTS_COUNT))
43 {
44 fputs("ERROR: Could not initialize key press event manager", stderr);
45 goto fail_initialization;
46 }
47
48 if (!initializeEventManager(&gKeyReleaseEventManager, MAX_KEY_LISTENER_LISTS_COUNT))
49 {
50 fputs("ERROR: Could not initialize key release event manager", stderr);
51 goto fail_initialization;
52 }
53
54 if (!initializeEventManager(&gMouseEventManager, MOUSE_LAST))
55 {
56 fputs("ERROR: Could not initialize mouse event manager", stderr);
57 goto fail_initialization;
58 }
59
60 const int initJoystickResult = initializeJoystickEventHandler();
61
62 if (initJoystickResult != NOT_FOUND_JOYSTICK && !initJoystickResult)
63 {
64 fputs("ERROR: Could not initialize joystick events handler\n", stderr);
65 goto fail_initialization;
66 }
67
68 goto success_initalization;
69
70fail_initialization:
71 freeEventManager(&gKeyPressEventManager, true);
72 freeEventManager(&gKeyReleaseEventManager, true);
73 freeEventManager(&gMouseEventManager, true);
74 closeJoystickEventHandler();
75 return false;
76
77success_initalization:
78 return true;
79}

105void loopEventHandlerApi()
106{
107 SDL_Event event;
108 do
109 {
123 if (pauseGameLoopForRendering)
124 {
125 SDL_UpdateWindowSurface(gGameEngineGraphicEnvironment.window);
126 resumeGameloop();
127 }
128
129 while (SDL_PollEvent(&event))
130 {
131 switch (event.type)
132 {
133 case SDL_KEYDOWN:
134 {
135 PROCESS_KEYDOWN(event);
136 break;
137 }
138 case SDL_KEYUP:
139 {
140 PROCESS_KEYUP(event);
141 break;
142 }
143 case SDL_MOUSEMOTION:
144 {
145 PROCESS_MOUSEMOTION(event);
146 break;
147 }
148 case SDL_MOUSEWHEEL:
149 {
150 if (event.wheel.y > 0)
151 {
152 push_wheel_event(SDL_MOUSEBUTTONDOWN, MOUSE_WHEEL_UP_BUTTON_PRESS);
153 push_wheel_event(SDL_MOUSEBUTTONUP, MOUSE_WHEEL_UP_BUTTON_PRESS);
154 }
155 else if (event.wheel.y < 0)
156 {
157 push_wheel_event(SDL_MOUSEBUTTONDOWN, MOUSE_WHEEL_DOWN_BUTTON_PRESS);
158 push_wheel_event(SDL_MOUSEBUTTONUP, MOUSE_WHEEL_DOWN_BUTTON_PRESS);
159 }
160 if (event.wheel.x < 0)
161 {
162 push_wheel_event(SDL_MOUSEBUTTONDOWN, TOUCH_LEFT_BUTTON_PRESS);
163 push_wheel_event(SDL_MOUSEBUTTONUP, TOUCH_LEFT_BUTTON_PRESS);
164 }
165 else if (event.wheel.x > 0)
166 {
167 push_wheel_event(SDL_MOUSEBUTTONDOWN, TOUCH_RIGHT_BUTTON_PRESS);
168 push_wheel_event(SDL_MOUSEBUTTONUP, TOUCH_RIGHT_BUTTON_PRESS);
169 }
170 break;
171 }
172 case SDL_MOUSEBUTTONDOWN:
173 {
174 PROCESS_MOUSEBUTTONDOWN(event);
175 break;
176 }
177 case SDL_MOUSEBUTTONUP:
178 {
179 PROCESS_MOUSEBUTTONUP(event);
180 break;
181 }
182 case SDL_QUIT:
183 {
184 PROCESS_QUIT(event);
185 break;
186 }
187 default:
188 handleJoystickEvents(&event);
189 break;
190 }
191 }
192
193 // Trigger events for all joystick buttons still pressed
194 TRIGGER_PRESSED_JOYSTICKS_BUTTONS();
195
196 // Trigger events for all keys still pressed
197 TRIGGER_PRESSED_KEYS();
198
199 // Trigger events for all mouse buttons still pressed
200 TRIGGER_PRESSED_MOUSE_BUTTONS();
201 } while (isGameRunning);
202}
