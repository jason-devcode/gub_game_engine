201static inline MeshGroup *readWavefrontObj(const char *filename)
202{
203 if (filename == NULL)
204 return NULL;
205
206 FILE *objFile = fopen(filename, "r");
207
208 if (objFile == NULL)
209 return NULL;
210
211 VertexLinearList *vertices = createVertexLinearList();
212 VertexTextureLinearList *verticesTexture = createVertexTextureLinearList();
213 VertexNormalLinearList *normals = createVertexNormalLinearList();
214 FaceLinearList *faces = createFaceLinearList();
215 MeshLinearList *meshes = createMeshLinearList();
216
217 Mesh3D *currentMesh = createMesh(); // Create mesh by default if there is not o in file
218
219 MeshGroup *meshGroup = createEmptyMeshGroup();
220
221 char currentFileLine[MAX_BUFFER_LINE] = {0};
222
223 while (fgets(currentFileLine, MAX_BUFFER_LINE, objFile) != NULL)
224 {
225 uint16_t lineKeyValue = *(uint16_t *)currentFileLine;
226
227 switch (lineKeyValue)
228 {
229 case KEY_READ_VERTEX_LINE:
230 parseVertexLine(vertices, currentFileLine);
231 break;
232 case KEY_READ_VERTEX_TEXTURE_LINE:
233 parseVertexTextureLine(verticesTexture, currentFileLine);
234 break;
235 case KEY_READ_VERTEX_NORMAL_LINE:
236 parseVertexNormalLine(normals, currentFileLine);
237 break;
238 case KEY_READ_FACE_LINE:
239 parseFaceLine(faces, currentFileLine);
240 break;
241 case KEY_READ_OBJECT_LINE:
242 UPDATE_AND_RESET_MESH(currentMesh, meshes, faces);
243
244 faces = createFaceLinearList();
245 currentMesh = createMesh();
246 parseObjectLine(currentMesh, currentFileLine);
247 break;
248 default:
249 break;
250 }
251 }
252 UPDATE_AND_RESET_MESH(currentMesh, meshes, faces);
253
254 meshGroup->meshes = convertMeshListToArray(meshes);
255 meshGroup->meshCount = meshes->countMeshes;
256
257 meshGroup->vertices = convertVertexListToArray(vertices);
258 meshGroup->vertexCount = vertices->countVertices;
259
260 meshGroup->vertices_texture = convertTextureListToArray(verticesTexture);
261 meshGroup->vertexTextureCount = verticesTexture->countTextures;
262
263 meshGroup->normals = convertNormalListToArray(normals);
264 meshGroup->normalsCount = normals->countNormals;
265
266 freeVertexLinearList(vertices);
267 freeVertexNormalLinearList(normals);
268 freeVertexTextureLinearList(verticesTexture);
269 freeMeshLinearList(meshes, false);
270
271 fclose(objFile);
272 objFile = NULL;
273
274 return meshGroup;
275}
