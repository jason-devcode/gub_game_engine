63static void updateCameraDirection()
64{
65 cosineCameraPitch = fast_cosine(cameraPitch), sineCameraPitch = fast_sine(cameraPitch);
66 cosineCameraYaw = fast_cosine(cameraYaw), sineCameraYaw = fast_sine(cameraYaw);
67 cosineCameraRoll = fast_cosine(cameraRoll), sineCameraRoll = fast_sine(cameraRoll);
68
69 double tempX = 0.0, X = 0;
70 double tempY = 0.0, Y = 0;
71 double tempZ = 1.0, Z = 0;
72
73 // rotate X Axis - ZY plane pitch
74 Z = tempZ * cosineCameraPitch - tempY * sineCameraPitch;
75 Y = tempY * cosineCameraPitch + tempZ * sineCameraPitch;
76 tempZ = Z;
77 tempY = Y;
78
79 // rotate Y Axis - XZ plane yaw
80 X = tempX * cosineCameraYaw + tempZ * sineCameraYaw;
81 Z = tempZ * cosineCameraYaw - tempX * sineCameraYaw;
82 tempX = X;
83 tempZ = Z;
84
85 // rotate Z Axis - XY plane roll
86 X = tempX * cosineCameraRoll - tempY * sineCameraRoll;
87 Y = tempY * cosineCameraRoll + tempX * sineCameraRoll;
88 tempX = X;
89 tempY = Y;
90
91 cameraDirection.x = tempX;
92 cameraDirection.y = tempY;
93 cameraDirection.z = tempZ;
94}
