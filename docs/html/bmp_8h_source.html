66Raster8888 *loadBMP(const char *filename)
67{
68 FILE *file = fopen(filename, "rb");
69 if (!file)
70 {
71 perror("Error opening file");
72 return NULL;
73 }
74
75 BMPFileHeader fileHeader;
76 BMPInfoHeader infoHeader;
77
78 fread(&fileHeader, sizeof(BMPFileHeader), 1, file);
79 fread(&infoHeader, sizeof(BMPInfoHeader), 1, file);
80
81 // Check if the file is a BMP
82 if (fileHeader.bfType != 0x4D42) // 'BM' in little-endian
83 {
84 fclose(file);
85 fprintf(stderr, "Not a BMP file\n");
86 return NULL;
87 }
88
89 // Check for supported bit depths
90 if (infoHeader.biBitCount != BMP_BIT_DEPTH_24 && infoHeader.biBitCount != BMP_BIT_DEPTH_32)
91 {
92 fclose(file);
93 fprintf(stderr, "Unsupported BMP bit depth: %d\n", infoHeader.biBitCount);
94 return NULL;
95 }
96
97 // Allocate memory for the raster
98 Raster8888 *raster = (Raster8888 *)malloc(sizeof(Raster8888));
99 if (!raster)
100 {
101 fclose(file);
102 fprintf(stderr, "Memory allocation failed\n");
103 return NULL;
104 }
105
106 raster->width = infoHeader.biWidth;
107 raster->height = abs(infoHeader.biHeight); // Height can be negative
108 raster->totalPixels = raster->width * raster->height;
109 raster->pixels = (uint32_t *)malloc(raster->totalPixels * sizeof(uint32_t));
110 if (!raster->pixels)
111 {
112 free(raster);
113 fclose(file);
114 fprintf(stderr, "Memory allocation failed\n");
115 return NULL;
116 }
117
118 // Move the file pointer to the beginning of the pixel data
119 fseek(file, fileHeader.bfOffBits, SEEK_SET);
120
121 // Calculate the padded row size
122 int row_padded = ((infoHeader.biWidth * infoHeader.biBitCount + 31) / 32) * 4;
123 uint8_t *row = (uint8_t *)malloc(row_padded);
124
125 if (!row)
126 {
127 free(raster->pixels);
128 free(raster);
129 fclose(file);
130 fprintf(stderr, "Memory allocation failed\n");
131 return NULL;
132 }
133
134 // Read the pixel data row by row
135 for (int y = 0; y < raster->height; y++)
136 {
137 int row_index = (infoHeader.biHeight > 0) ? (raster->height - y - 1) : y; // Handle vertical orientation
138
139 fread(row, 1, row_padded, file);
140
141 for (int x = 0; x < raster->width; x++)
142 {
143 uint32_t pixel = 0;
144
145 if (infoHeader.biBitCount == BMP_BIT_DEPTH_24)
146 {
147 // Convert 24-bit BMP to 32-bit RGBA
148 pixel |= row[x * 3 + 2] << BMP_RED_SHIFT; // Red
149 pixel |= row[x * 3 + 1] << BMP_GREEN_SHIFT; // Green
150 pixel |= row[x * 3 + BMP_BLUE_SHIFT]; // Blue
151 pixel |= BMP_ALPHA_MASK; // Alpha (opaque)
152 }
153 else if (infoHeader.biBitCount == BMP_BIT_DEPTH_32)
154 {
155 // Read 32-bit BMP directly to 32-bit RGBA
156 pixel |= row[x * 4 + 2] << BMP_RED_SHIFT; // Red
157 pixel |= row[x * 4 + 1] << BMP_GREEN_SHIFT; // Green
158 pixel |= row[x * 4 + BMP_BLUE_SHIFT]; // Blue
159 pixel |= row[x * 4 + 3] << BMP_ALPHA_SHIFT; // Alpha
160 }
161
162 raster->pixels[row_index * raster->width + x] = pixel;
163 }
164 }
165
166 free(row);
167 fclose(file);
168 return raster;
169}

178int saveBMP(const char *filename, const Raster8888 *raster)
179{
180 FILE *file = fopen(filename, "wb");
181 if (!file)
182 {
183 perror("Error opening file");
184 return -1;
185 }
186
187 BMPFileHeader fileHeader = {0};
188 BMPInfoHeader infoHeader = {0};
189
190 fileHeader.bfType = 0x4D42;
191 fileHeader.bfSize = sizeof(BMPFileHeader) + sizeof(BMPInfoHeader) + (raster->totalPixels * sizeof(uint32_t));
192 fileHeader.bfOffBits = sizeof(BMPFileHeader) + sizeof(BMPInfoHeader);
193
194 infoHeader.biSize = sizeof(BMPInfoHeader);
195 infoHeader.biWidth = raster->width;
196 infoHeader.biHeight = -raster->height; // Negative height to vertical flip
197 infoHeader.biPlanes = 1;
198 infoHeader.biBitCount = 32; // Assuming 32 bits per pixel (RGBA)
199 infoHeader.biCompression = 0; // No compression
200 infoHeader.biSizeImage = raster->totalPixels * sizeof(uint32_t);
201
202 fwrite(&fileHeader, sizeof(BMPFileHeader), 1, file);
203 fwrite(&infoHeader, sizeof(BMPInfoHeader), 1, file);
204 fwrite(raster->pixels, sizeof(uint32_t), raster->totalPixels, file);
205
206 fclose(file);
207 return 0;
208}
