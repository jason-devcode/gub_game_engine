39static inline uint32_t applyGammaCorrection(uint32_t color, double gammaFactor)
40{
41 double R = (double)GETR(color) / 255.0; // Decimal representation of the red channel [0-1]
42 double G = (double)GETG(color) / 255.0; // Decimal representation of the green channel [0-1]
43 double B = (double)GETB(color) / 255.0; // Decimal representation of the blue channel [0-1]
44
45 uint8_t newR = (uint8_t)(pow(R, 1 / gammaFactor) * 255.0);
46 uint8_t newG = (uint8_t)(pow(G, 1 / gammaFactor) * 255.0);
47 uint8_t newB = (uint8_t)(pow(B, 1 / gammaFactor) * 255.0);
48
49 return RGB(newR, newG, newB);
50}
