140{

142 register int DY = y2 - y1;
143

145 register int absDY = fast_abs(DY);
146
147 register double steps = (double)(absDX > absDY ? absDX : absDY);
148 steps += !steps; // protect for zero division

244static inline void drawRectangleFill(int32_t x1, int32_t y1, int32_t x2, int32_t y2, uint32_t color)
245{
246 // Normalize coordinates if necessary
247 if (x1 > x2)
248 fast_swap_bitwise(x1, x2);
249 if (y1 > y2)
250 fast_swap_bitwise(y1, y2);

252 // Calculate width and height from normalized x1, y1, x2, y2
253 int32_t width = x2 - x1 + 1;
254 int32_t height = y2 - y1 + 1;
255
256 // Protect against invalid dimensions and out-of-bounds issues
257 if (framebuffer == NULL || width < 0 || height < 0 || !(x1 < (int)gScreenWidth) || !(y1 < (int)gScreenHeight))
258 return;
259
260 // Protect against drawing outside the left and top boundaries
261 if (x2 < 0)
262 return;
263 if (y2 < 0)
264 return;
265
266 // Calculate correct values in relation to screen edges collisions
267 uint32_t correctWidth = width;
268 uint32_t correctHeight = height;
269
270 correctWidth -= (gScreenWidth < (x2 + 1)) ? (x2 + 1) - gScreenWidth : 0; // Adjust for right edge collision
271 correctWidth += (x1 < 0) ? x1 : 0; // Adjust for left edge collision
272
273 correctHeight -= (gScreenHeight < (y2 + 1)) ? (y2 + 1) - gScreenHeight : 0; // Adjust for bottom edge collision
274 correctHeight += (y1 < 0) ? y1 : 0; // Adjust for top edge collision
275
276 uint32_t correctX1 = x1 < 0 ? 0 : x1;
277 uint32_t correctY1 = y1 < 0 ? 0 : y1;
278 uint32_t correctX2 = correctX1 + correctWidth - 1;
279 uint32_t correctY2 = correctY1 + correctHeight - 1;

282 register uint32_t *beginPixelsRowIterator = &framebuffer[calculatePixelIndex(0, correctY1, gScreenWidth)];

284

286
287 register uint32_t pixelsRowWidth = gScreenWidth;

317{

613static inline void drawBezierNPoints(const Vec2f *controlPoints, int numControlPoints, int numSteps, unsigned int color)
614{
615 float stepSize = 1.0f / numSteps;
616 Vec2f currentPoint, previousPoint;
617
618 for (int currentStep = 0; currentStep <= numSteps; ++currentStep)
619 {
620 float t = currentStep * stepSize;
621 currentPoint = (Vec2f){0, 0};
622
623 for (int controlPointIndex = 0; controlPointIndex < numControlPoints; ++controlPointIndex)
624 {
625 float bernsteinCoefficient = gPascalTriangle[numControlPoints - 1][controlPointIndex] *
626 pow(t, controlPointIndex) *
627 pow(1 - t, numControlPoints - 1 - controlPointIndex);
628
629 currentPoint.x += bernsteinCoefficient * controlPoints[controlPointIndex].x;
630 currentPoint.y += bernsteinCoefficient * controlPoints[controlPointIndex].y;
631 }
632
633 if (currentStep > 0)
634 drawLine(previousPoint.x, previousPoint.y, currentPoint.x, currentPoint.y, color);
635 previousPoint = currentPoint;
636 }
637}
