50// Function to load TGA image
51Raster8888 *loadTGA(const char *filename)
52{
53 FILE *file = fopen(filename, "rb");
54 if (!file)
55 {
56 perror("Error opening file");
57 return NULL;
58 }
59
60 TGAHeader header;
61 if (fread(&header, sizeof(TGAHeader), 1, file) != 1)
62 {
63 handleError("Error reading TGA header", file, NULL, NULL);
64 return NULL;
65 }
66
67 if (header.imageType != 2 && header.imageType != 10)
68 { // Only supporting uncompressed RGB and RLE RGB
69 handleError("Unsupported TGA image type", file, NULL, NULL);
70 return NULL;
71 }
72
73 Raster8888 *raster = (Raster8888 *)malloc(sizeof(Raster8888));
74 if (!raster)
75 {
76 handleError("Memory allocation failed", file, NULL, NULL);
77 return NULL;
78 }
79
80 raster->width = header.width;
81 raster->height = header.height;
82 raster->totalPixels = raster->width * raster->height;
83 raster->pixels = (uint32_t *)malloc(raster->totalPixels * sizeof(uint32_t));
84 if (!raster->pixels)
85 {
86 handleError("Memory allocation failed", file, NULL, raster);
87 return NULL;
88 }
89
90 int result = 0;
91 if (header.imageType < sizeof(readPixelData) / sizeof(readPixelData[0]) && readPixelData[header.imageType])
92 {
93 result = readPixelData[header.imageType](file, raster);
94 }
95 else
96 {
97 handleError("Unsupported TGA image type", file, NULL, raster);
98 return NULL;
99 }
100
101 if (result != 0)
102 {
103 handleError("Error reading TGA pixel data", file, NULL, raster);
104 return NULL;
105 }
106
107 fclose(file);
108 return raster;
109}

111// Function to read uncompressed RGB pixel data
112static int readUncompressedRGB(FILE *file, Raster8888 *raster)
113{
114 size_t pixelDataSize = raster->totalPixels * 3;
115 uint8_t *tempPixels = (uint8_t *)malloc(pixelDataSize);
116 if (!tempPixels)
117 {
118 return -1;
119 }
120
121 if (fread(tempPixels, pixelDataSize, 1, file) != 1)
122 {
123 free(tempPixels);
124 return -1;
125 }
126
127 for (size_t i = 0; i < raster->totalPixels; ++i)
128 {
129 uint8_t r = tempPixels[i * 3 + 0];
130 uint8_t g = tempPixels[i * 3 + 1];
131 uint8_t b = tempPixels[i * 3 + 2];
132 raster->pixels[i] = (255 << 24) | (r << 16) | (g << 8) | b; // Set alpha to 255
133 }
134
135 free(tempPixels);
136 return 0;
137}

139// Function to read compressed RGB pixel data (RLE)
140static int readCompressedRGB(FILE *file, Raster8888 *raster)
141{
142 size_t pixelIndex = 0;
143 while (pixelIndex < raster->totalPixels)
144 {
145 uint8_t headerByte;
146 if (fread(&headerByte, sizeof(uint8_t), 1, file) != 1)
147 {
148 return -1;
149 }
150
151 bool isRLE = (headerByte & 0x80) != 0;
152 uint8_t count = (headerByte & 0x7F) + 1;
153
154 if (isRLE)
155 {
156 uint8_t rgb[3];
157 if (fread(rgb, sizeof(uint8_t), 3, file) != 3)
158 {
159 return -1;
160 }
161
162 for (uint8_t i = 0; i < count; ++i)
163 {
164 if (pixelIndex >= raster->totalPixels)
165 break;
166 raster->pixels[pixelIndex++] = (255 << 24) | (rgb[0] << 16) | (rgb[1] << 8) | rgb[2];
167 }
168 }
169 else
170 {
171 uint8_t rgb[3 * 128]; // Max run length of 128
172 if (fread(rgb, sizeof(uint8_t), 3 * count, file) != 3 * count)
173 {
174 return -1;
175 }
176
177 for (uint8_t i = 0; i < count; ++i)
178 {
179 if (pixelIndex >= raster->totalPixels)
180 break;
181 raster->pixels[pixelIndex++] = (255 << 24) | (rgb[i * 3 + 0] << 16) | (rgb[i * 3 + 1] << 8) | rgb[i * 3 + 2];
182 }
183 }
184 }
185
186 return 0;
187}

189// Function to save Raster8888 as a TGA file
190int saveTGA(const char *filename, const Raster8888 *raster)
191{
192 FILE *file = fopen(filename, "wb");
193 if (!file)
194 {
195 perror("Error opening file");
196 return -1;
197 }
198
199 TGAHeader header = {0};
200 header.idLength = 0;
201 header.colorMapType = 0;
202 header.imageType = 2; // Uncompressed RGB
203 header.width = raster->width;
204 header.height = raster->height;
205 header.pixelDepth = 32; // 32 bits per pixel (RGBA)
206 header.imageDescriptor = 0;
207
208 if (fwrite(&header, sizeof(TGAHeader), 1, file) != 1)
209 {
210 fclose(file);
211 fprintf(stderr, "Error writing TGA header\n");
212 return -1;
213 }
214
215 size_t pixelDataSize = raster->totalPixels * sizeof(uint32_t);
216 uint8_t *tempPixels = (uint8_t *)malloc(pixelDataSize);
217 if (!tempPixels)
218 {
219 fclose(file);
220 fprintf(stderr, "Memory allocation failed\n");
221 return -1;
222 }
223
224 for (size_t i = 0; i < raster->totalPixels; ++i)
225 {
226 tempPixels[i * 4 + 0] = (raster->pixels[i] >> 16) & 0xFF; // Red
227 tempPixels[i * 4 + 1] = (raster->pixels[i] >> 8) & 0xFF; // Green
228 tempPixels[i * 4 + 2] = raster->pixels[i] & 0xFF; // Blue
229 tempPixels[i * 4 + 3] = (raster->pixels[i] >> 24) & 0xFF; // Alpha
230 }
231
232 if (fwrite(tempPixels, pixelDataSize, 1, file) != 1)
233 {
234 free(tempPixels);
235 fclose(file);
236 fprintf(stderr, "Error writing TGA pixel data\n");
237 return -1;
238 }
239
240 free(tempPixels);
241 fclose(file);
242 return 0;
243}
