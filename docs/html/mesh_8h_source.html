174static inline void releaseMeshGroupResources(MeshGroup *group)
175{
176 if (!group)
177 return;
178
179 // Free the allocated memory for vertices, normals, textures
180 if (group->vertices)
181 free(group->vertices);
182 if (group->normals)
183 free(group->normals);
184 if (group->vertices_texture)
185 free(group->vertices_texture);
186
187 // Free each mesh within the group
188 if (group->meshes)
189 {
190 for (int i = 0; i < group->meshCount; ++i)
191 {
192 releaseMesh3DResources(&group->meshes[i]);
193 }
194 free(group->meshes);
195 }
196
197 // Set pointers to NULL after freeing
198 group->vertices = NULL;
199 group->normals = NULL;
200 group->vertices_texture = NULL;
201 group->meshes = NULL;

214MeshGroup *mergeMeshGroups(MeshGroup **groups, int groupCount)
215{
216 if (!groups || groupCount <= 0)
217 {
218 fprintf(stderr, "Invalid input for merging mesh groups.\n");
219 return NULL;
220 }
221
222 MeshGroup *mergedGroup = createEmptyMeshGroup();
223
224 // Calculate total counts
225 int totalVertexCount = 0;
226 int totalNormalCount = 0;
227 int totalTextureCount = 0;
228 int totalMeshCount = 0;
229
230 for (int i = 0; i < groupCount; ++i)
231 {
232 if (groups[i])
233 {
234 totalVertexCount += groups[i]->vertexCount;
235 totalNormalCount += groups[i]->normalsCount;
236 totalTextureCount += groups[i]->vertexTextureCount;
237 totalMeshCount += groups[i]->meshCount;
238 }
239 }
240
241 // Allocate memory for the merged group
242 mergedGroup->vertices = (Vec3f *)malloc(totalVertexCount * sizeof(Vec3f));
243 mergedGroup->normals = (Vec3f *)malloc(totalNormalCount * sizeof(Vec3f));
244 mergedGroup->vertices_texture = (Vec2f *)malloc(totalTextureCount * sizeof(Vec2f));
245 mergedGroup->meshes = (Mesh3D *)malloc(totalMeshCount * sizeof(Mesh3D));
246
247 if (!mergedGroup->vertices || !mergedGroup->normals || !mergedGroup->vertices_texture || !mergedGroup->meshes)
248 {
249 fprintf(stderr, "Error allocating memory for merged mesh group.\n");
250 releaseMeshGroupResources(mergedGroup);
251 return NULL;
252 }
253
254 // Copy data and adjust indices
255 int vertexOffset = 0;
256 int normalOffset = 0;
257 int textureOffset = 0;
258 int meshOffset = 0;
259
260 for (int i = 0; i < groupCount; ++i)
261 {
262 MeshGroup *group = groups[i];
263 if (!group)
264 continue;
265
266 // Copy vertices, normals, and textures
267 memcpy(mergedGroup->vertices + vertexOffset, group->vertices, group->vertexCount * sizeof(Vec3f));
268 memcpy(mergedGroup->normals + normalOffset, group->normals, group->normalsCount * sizeof(Vec3f));
269 memcpy(mergedGroup->vertices_texture + textureOffset, group->vertices_texture, group->vertexTextureCount * sizeof(Vec2f));
270
271 // Copy and adjust meshes
272 for (int j = 0; j < group->meshCount; ++j)
273 {
274 Mesh3D *sourceMesh = &group->meshes[j];
275 Mesh3D *targetMesh = &mergedGroup->meshes[meshOffset + j];
276
277 *targetMesh = *sourceMesh;
278 targetMesh->facesIndices = (FaceIndices *)malloc(sourceMesh->facesCount * sizeof(FaceIndices));
279
280 if (!targetMesh->facesIndices)
281 {
282 fprintf(stderr, "Error allocating memory for facesIndices in merged mesh group.\n");
283 releaseMeshGroupResources(mergedGroup);
284 return NULL;
285 }
286
287 memcpy(targetMesh->facesIndices, sourceMesh->facesIndices, sourceMesh->facesCount * sizeof(FaceIndices));
288
289 // Adjust indices
290 for (int k = 0; k < sourceMesh->facesCount; ++k)
291 {
292 targetMesh->facesIndices[k].vertexA += vertexOffset;
293 targetMesh->facesIndices[k].vertexB += vertexOffset;
294 targetMesh->facesIndices[k].vertexC += vertexOffset;
295
296 targetMesh->facesIndices[k].vertexNormalA += normalOffset;
297 targetMesh->facesIndices[k].vertexNormalB += normalOffset;
298 targetMesh->facesIndices[k].vertexNormalC += normalOffset;
299
300 targetMesh->facesIndices[k].vertexTextureA += textureOffset;
301 targetMesh->facesIndices[k].vertexTextureB += textureOffset;
302 targetMesh->facesIndices[k].vertexTextureC += textureOffset;
303 }
304 }
305
306 vertexOffset += group->vertexCount;
307 normalOffset += group->normalsCount;
308 textureOffset += group->vertexTextureCount;
309 meshOffset += group->meshCount;
310 }
311
312 mergedGroup->vertexCount = totalVertexCount;
313 mergedGroup->normalsCount = totalNormalCount;
314 mergedGroup->vertexTextureCount = totalTextureCount;
315 mergedGroup->meshCount = totalMeshCount;
316
317 return mergedGroup;
318}
